async function extractVideos() { const firebaseUrl = "https://quest-full-default-rtdb.firebaseio.com/.json"; try { const response = await fetch(firebaseUrl); const data = await response.json(); // تحويل كل البيانات لنص للبحث السريع عن روابط const allDataText = JSON.stringify(data); // البحث عن أي رابط ينتهي بـ .m3u8 أو يحتوي على vimeo const videoRegex = /(https?:\/\/[^\s"]+\.(m3u8|mp4)|https?:\/\/player\.vimeo\.com[^\s"]+)/g; const links = allDataText.match(videoRegex); if (links) { console.log("✅ لقينا الكنز! دي روابط الفيديوهات المباشرة:"); console.table([...new Set(links)]); // عرض الروابط بدون تكرار } else { console.log("❌ مفيش روابط فيديوهات واضحة في النود ده."); } } catch (e) { console.log("⚠️ السيرفر محمي بـ Rules تمنع القراءة الكاملة."); } }
